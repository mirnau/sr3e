### Task 1: Create ShoppingCart Injection Component

**Action**: Create shopping cart icon component with dual-purpose toggle logic (terminate creation OR toggle karma shopping).

**Files**:
- Create `module/ui/actors/injections/ShoppingCart.svelte`

**Implementation**:
```svelte
<script lang="ts">
   import { StoreManager } from "@helpers/StoreManager.svelte";
   import { CreationPointsService } from "@services/character-creation/CreationPointsService";
   import { FLAGS } from "@constants/flags";

   const { actor, config } = $props<{ actor: Actor; config: any }>();
   const storeManager = StoreManager.Subscribe(actor);
   const isCharacterCreation = storeManager.GetFlagStore(FLAGS.ACTOR.IS_CHARACTER_CREATION);
   const isShoppingState = storeManager.GetFlagStore(FLAGS.ACTOR.IS_SHOPPING_STATE);

   async function handleToggle() {
      if ($isCharacterCreation) {
         await terminateCreation();
      } else {
         await isShoppingState.update(v => !v);
      }
   }

   async function terminateCreation() {
      const service = CreationPointsService.Instance();
      const hasUnspentPoints = service.hasUnspentPoints(actor);

      if (hasUnspentPoints) {
         const confirmed = await showWarningDialog();
         if (!confirmed) return;
      }

      await isCharacterCreation.update(() => false);
   }

   async function showWarningDialog(): Promise<boolean> {
      return Dialog.confirm({
         title: "Finish Character Creation?",
         content: "<p>You have unspent creation points. These will be discarded.</p><p>Are you sure you want to finish creation?</p>",
         yes: () => true,
         no: () => false,
      });
   }
</script>

<div>
   <button
      type="button"
      class={`header-control icon fa-solid fa-cart-shopping ${$isShoppingState || $isCharacterCreation ? "pulsing-green-cart" : ""}`}
      onclick={handleToggle}
   />
</div>
```

**Verification**:
- Icon appears in header
- Pulses when creation mode OR karma shopping active
- Clicking during creation shows warning if points unspent
- Clicking during creation clears flag if confirmed
- Clicking post-creation toggles karma shopping

### Task 4: Create CharacterCreationManager Injection Component

**Action**: Create manager component that switches between attribute and skill point displays based on creation phase.

**Files**:
- Create `module/ui/actors/injections/CharacterCreationManager.svelte`

**Implementation**:
```svelte
<script lang="ts">
   import AttributePointsState from "./AttributePointsState.svelte";
   import SkillPointsState from "./SkillPointsState.svelte";
   import { StoreManager } from "@helpers/StoreManager.svelte";
   import { FLAGS } from "@constants/flags";

   const { actor, config } = $props<{ actor: Actor; config: any }>();
   const storeManager = StoreManager.Subscribe(actor);

   const attributeAssignmentLocked = storeManager.GetFlagStore(FLAGS.ACTOR.ATTRIBUTE_ASSIGNMENT_LOCKED);
   const isCharacterCreation = storeManager.GetFlagStore(FLAGS.ACTOR.IS_CHARACTER_CREATION);
</script>

{#if $isCharacterCreation}
   <div class="character-creation-manager">
      {#if $attributeAssignmentLocked}
         <SkillPointsState {actor} {config} />
      {:else}
         <AttributePointsState {actor} {config} />
      {/if}
   </div>
{/if}

<style>
   .character-creation-manager {
      margin-bottom: 12px;
   }
</style>
```

**Verification**:
- Only visible when `isCharacterCreation = true`
- Shows AttributePointsState when attributes not locked
- Shows SkillPointsState when attributes locked
- Component switches reactively when lock flag changes

### Task 5: Integrate Injections into CharacterSheet

**Action**: Add injection methods to CharacterSheet.ts for shopping cart and creation manager.

**Files**:
- Modify `module/sheets/actors/CharacterSheet.ts`

**Implementation**:
```typescript
import { mount } from "svelte";
import ShoppingCart from "../../ui/actors/injections/ShoppingCart.svelte";
import CharacterCreationManager from "../../ui/actors/injections/CharacterCreationManager.svelte";

protected _replaceHTML(_result: unknown, windowContent: HTMLElement, _options: DeepPartial<RenderOptions>): void {
   this._unmountAllApps();
   const form = this.form as HTMLFormElement;

   // Mount main sheet
   this.#app = mount(CharacterSheetApp, {
      target: windowContent,
      props: { actor: this.document as Actor, form }
   });
   this.apps.push(this.#app);

   // Inject into header
   const header = form.querySelector("header.window-header");
   this._injectNeonName(header);
   this._injectNewsFeed(header);
   this._injectShoppingCart(header); // NEW

   // Inject creation manager into main content
   const mainContent = form.querySelector(".window-content");
   this._injectCreationManager(mainContent); // NEW
}

private _injectShoppingCart(header: Element | null): void {
   if (!header) return;

   const container = document.createElement("div");
   header.appendChild(container);

   const app = mount(ShoppingCart, {
      target: container,
      props: { actor: this.document as Actor, config: CONFIG.SR3E }
   });
   this.apps.push(app);
}

private _injectCreationManager(content: Element | null): void {
   if (!content) return;

   const container = document.createElement("div");
   content.insertBefore(container, content.firstChild);

   const app = mount(CharacterCreationManager, {
      target: container,
      props: { actor: this.document as Actor, config: CONFIG.SR3E }
   });
   this.apps.push(app);
}
```

**Verification**:
- Shopping cart icon appears in header
- Creation manager appears at top of sheet content
- Both components unmount correctly on close
- No TypeScript errors
